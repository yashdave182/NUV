# AgriTech Frontend Implementation Guide

A production-ready frontend specification for the TetraTHON 2026 AgriTech API.

## 🎯 Core Requirements

| Requirement | Implementation |
|-------------|----------------|
| **PWA-first** | Service Worker, offline caching, installable |
| **Offline-capable** | IndexedDB + Background Sync + React Query |
| **19 Indian languages** | i18next + native scripts + RTL (Urdu) |
| **SMS share** | Web Share API + clipboard fallback |
| **TTS/Voice** | Web Speech API (listen to advisories) |
| **Maps** | MapLibre GL + offline tiles |
| **Charts** | Recharts (yield, price trends, NDVI) |
| **Phone auth** | OTP-based (10-digit Indian mobile) |

  baseURL: import.meta.env.VITE_API_URL || 'https://api.agritech.in',
  timeout: 15000,
  headers: { 'Content-Type': 'application/json' },
});

// Request interceptor - add auth token
api.interceptors.request.use((config) => {
  const token = useAuthStore.getState().token;
  if (token) config.headers.Authorization = `Bearer ${token}`;
  config.headers['Accept-Language'] = useAuthStore.getState().language || 'gu';
  return config;
});

// Response interceptor - handle offline queue
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (!navigator.onLine && error.config?.method === 'POST') {
      // Queue for background sync
      await queueForSync(error.config);
      return Promise.reject({ ...error, queued: true });
    }
    return Promise.reject(error);
  }
);

export const useAdvisory = (payload: FarmInputRequest) =>
  useMutation({
    mutationFn: () => api.post('/agriculture/advisory', payload),
    onSuccess: (data) => {

### Offline-First Hook

```typescript
// src/shared/hooks/useOffline.ts
import { useState, useEffect } from 'react';
import { openDB } from 'idb';

const DB_NAME = 'agritech-offline';
const STORE_ADVISORY = 'advisories';
const STORE_SYNC = 'sync-queue';

export function useOffline() {
  const [isOnline, setIsOnline] = useState(navigator.onLine);
  const [pendingCount, setPendingCount] = useState(0);

  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true);
      processSyncQueue();
    };
    const handleOffline = () => setIsOnline(false);

    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);

    updatePendingCount();
    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  async function updatePendingCount() {
    const db = await openDB(DB_NAME, 1);
    const count = await db.count(STORE_SYNC);
    setPendingCount(count);
  }

  async function queueForSync(request: any) {
    const db = await openDB(DB_NAME, 1);
    await db.add(STORE_SYNC, {
      ...request,
      timestamp: Date.now(),
      retries: 0,
    });
    updatePendingCount();
    
    // Register background sync
    if ('serviceWorker' in navigator && 'sync' in window.ServiceWorkerRegistration.prototype) {
      const registration = await navigator.serviceWorker.ready;
      await registration.sync.register('sync-api-requests');
    }
  }

  async function processSyncQueue() {
    const db = await openDB(DB_NAME, 1);
    const queue = await db.getAll(STORE_SYNC);
    
    for (const item of queue) {
      try {
        await api.request(item);
        await db.delete(STORE_SYNC, item.id);
      } catch (e) {
        if (item.retries < 3) {
          await db.put(STORE_SYNC, { ...item, retries: item.retries + 1 });
        }
      }
    }
    updatePendingCount();
  }

  async function cacheAdvisory(phone: string, data: any) {
    const db = await openDB(DB_NAME, 1);
    await db.put(STORE_ADVISORY, { phone, data, cachedAt: Date.now() });
  }

  async function getCachedAdvisory(phone: string) {
    const db = await openDB(DB_NAME, 1);
    return db.get(STORE_ADVISORY, phone);
  }

  return { isOnline, pendingCount, queueForSync, cacheAdvisory, getCachedAdvisory };
}
```

### TTS Hook

```typescript
// src/shared/hooks/useTTS.ts
import { useState, useCallback, useRef } from 'react';

const langMap: Record<string, string> = {
  hi: 'hi-IN', gu: 'gu-IN', mr: 'mr-IN', pa: 'pa-IN',
  bn: 'bn-IN', ta: 'ta-IN', te: 'te-IN', kn: 'kn-IN',
  or: 'or-IN', ur: 'ur-IN', en: 'en-IN',
  hinglish: 'hi-IN', gujlish: 'gu-IN', marathish: 'mr-IN',
  punjabish: 'pa-IN', tamlish: 'ta-IN', teluguish: 'te-IN',
  kannadish: 'kn-IN', bengalish: 'bn-IN',
};

export function useTTS(language: string) {
  const [isSpeaking, setIsSpeaking] = useState(false);
  const utteranceRef = useRef<SpeechSynthesisUtterance | null>(null);

  const speak = useCallback((text: string) => {
    if (!('speechSynthesis' in window)) return;
    
    stop();
    
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = langMap[language] || 'en-IN';
    utterance.rate = 0.9;
    utterance.pitch = 1;
    utterance.volume = 1;
    
    utterance.onstart = () => setIsSpeaking(true);
    utterance.onend = () => setIsSpeaking(false);
    utterance.onerror = () => setIsSpeaking(false);
    
    utteranceRef.current = utterance;
    window.speechSynthesis.speak(utterance);
  }, [language]);

  const stop = useCallback(() => {
    window.speechSynthesis.cancel();
    setIsSpeaking(false);
  }, []);

  return { speak, stop, isSpeaking };
}

// src/shared/utils/share.ts
export async function shareAdvisory(advisory: AdvisoryItem) {
  const text = `🌾 ${advisory.title}\n${advisory.description}\n\n${advisory.action_items.map((a, i) => `${i+1}. ${a}`).join('\n')}\n\n— AgriTech Advisory`;
  
  if (navigator.share) {
    try {
      await navigator.share({ title: 'AgriTech Advisory', text });
      return;
    } catch (e) { /* fallback */ }
  }
  
  // Clipboard fallback
  await navigator.clipboard.writeText(text);
  toast.success('Copied to clipboard! Send via SMS/WhatsApp');
  
  // Open SMS app
  if (/Android|iPhone/.test(navigator.userAgent)) {
    window.location.href = `sms:?body=${encodeURIComponent(text)}`;
  }
}

// src/features/weather/FarmMap.tsx
import Map from 'maplibre-gl';
import 'maplibre-gl/dist/maplibre-gl.css';
import { useEffect, useRef } from 'react';

interface FarmMapProps {
  latitude: number;
  longitude: number;
  ndviLayer?: GeoJSON.FeatureCollection;
  weatherAlerts?: WeatherAlert[];
}

export function FarmMap({ latitude, longitude, ndviLayer, weatherAlerts }: FarmMapProps) {
  const mapRef = useRef<HTMLDivElement>(null);
  const map = useRef<Map | null>(null);

  useEffect(() => {
    if (!mapRef.current || map.current) return;

    map.current = new Map({
      container: mapRef.current,
      style: 'https://demotiles.maplibre.org/style.json',
      center: [longitude, latitude],
      zoom: 14,
      pitch: 0,
    });

    map.current.addControl(new Map.NavigationControl(), 'top-right');
    map.current.addControl(new Map.GeolocateControl({
      positionOptions: { enableHighAccuracy: true },
      trackUserLocation: true,
    }), 'top-right');

    // Farm boundary
    map.current.on('load', () => {
      map.current!.addSource('farm', {
        type: 'geojson',
        data: {
          type: 'Feature',
          geometry: {
            type: 'Point',
            coordinates: [longitude, latitude],
          },
        },
      });
      map.current!.addLayer({
        id: 'farm-marker',
        type: 'circle',
        source: 'farm',
        paint: {
          'circle-radius': 12,
          'circle-color': '#1B5E20',
          'circle-stroke-width': 3,
          'circle-stroke-color': '#fff',
        },
      });

      // NDVI Layer
      if (ndviLayer) {
        map.current!.addSource('ndvi', { type: 'geojson', data: ndviLayer });
        map.current!.addLayer({
          id: 'ndvi-layer',
          type: 'fill',
          source: 'ndvi',
          paint: {
            'fill-color': [
              'interpolate', ['linear'], ['get', 'ndvi'],
              0, '#d32f2f', 0.2, '#f57c00', 0.4, '#fbc02d',
              0.6, '#8bc34a', 0.8, '#43a047', 1, '#1b5e20'
            ],
            'fill-opacity': 0.6,
          },
        });
      }

      // Weather alerts
      weatherAlerts?.forEach((alert, i) => {
        map.current!.addSource(`alert-${i}`, {
          type: 'geojson',
          data: { type: 'Feature', geometry: { type: 'Point', coordinates: [alert.lon, alert.lat] } },
        });
        map.current!.addLayer({
          id: `alert-${i}`,
          type: 'symbol',
          source: `alert-${i}`,
          layout: {
            'icon-image': 'warning',
            'icon-size': 1.2,
            'text-field': alert.type,
            'text-offset': [0, 1.5],
          },
        });
      });
    });

    return () => map.current?.remove();
  }, [latitude, longitude, ndviLayer, weatherAlerts]);

  return <div ref={mapRef} className="w-full h-96 rounded-xl" />;
}
```
